## 오늘의 학습 키워드

<br>

## 제출한 코드
```
#include <iostream>
#include <queue>
using namespace std;

struct CustomCompare
{
	bool operator()(int a, int b)
	{
		int absA = abs(a);
		int absB = abs(b);

		if (absA == absB)
		{
			return a > b;
		}

		return absA > absB;
	}
};

priority_queue<int, vector<int>, CustomCompare> list;

int main(void)
{
	int repeat, temp;

	cin >> repeat;

	for (int i = 0; i < repeat; i++)
	{
		cin >> temp;

		if (temp == 0)
		{
			if (list.empty())
			{
				cout << "0\n";
				continue;
			}

			cout << list.top() << '\n';
			list.pop();
			continue;
		}

		list.push(temp);
	}

	return 0;
}
```
